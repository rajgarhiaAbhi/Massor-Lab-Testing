function doGet(e) { var sheet = SpreadsheetApp.getActiveSheet(); var data = sheet.getDataRange().getValues(); if (data.length <= 1) return out({ status: 'ok', data: [] }); var headers = data[0]; var rows = data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); return out({ status: 'ok', data: rows }); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = {}; try { data = JSON.parse(e.postData.contents); } catch(_) {} var HEADERS = ['Date','Time','Size', 'Broken Final','Chaanti Rej1','Daal Rej1', 'Chaanti Rej2','Daal Rej2', 'Tokar1 Type','Tokar1 In','Tokar1 Out', 'Tokar2 Type','Tokar2 In','Tokar2 Out', 'Chaanti1 Type','Chaanti1 In','Chaanti1 Out', 'Chaanti2 Type','Chaanti2 In','Chaanti2 Out', 'Remarks']; // Ensure main sheet has headers var main = ss.getSheets()[0]; if (main.getLastRow() === 0) main.appendRow(HEADERS); // Add new type — create a sheet tab for it if (data.action === 'addType') { var tName = data.typeName; if (!ss.getSheetByName(tName)) { var ns = ss.insertSheet(tName); ns.appendRow(['Date','Time','Machine','Input','Output']); } return out({ status: 'ok' }); } // Rename type — rename existing sheet tab if (data.action === 'renameType') { var oldSheet = ss.getSheetByName(data.oldName); if (oldSheet) { oldSheet.setName(data.newName); } else { // Tab didn't exist yet — create it with new name var ns2 = ss.insertSheet(data.newName); ns2.appendRow(['Date','Time','Machine','Input','Output']); } return out({ status: 'ok' }); } // Add entry to main sheet if (data.action === 'add') { var r = data.entry; main.appendRow([r.date,r.time,r.size, r.brokenFinal,r.chaantiRej1,r.daalRej1, r.chaantiRej2,r.daalRej2, r.tokar1type,r.tokar1in,r.tokar1out, r.tokar2type,r.tokar2in,r.tokar2out, r.chaanti1type,r.chaanti1in,r.chaanti1out, r.chaanti2type,r.chaanti2in,r.chaanti2out, r.remarks]); // Also write to individual type sheets var typeEntries = [ [r.tokar1type, 'Tokar 1', r.tokar1in, r.tokar1out], [r.tokar2type, 'Tokar 2', r.tokar2in, r.tokar2out], [r.chaanti1type, 'Chaanti 1', r.chaanti1in, r.chaanti1out], [r.chaanti2type, 'Chaanti 2', r.chaanti2in, r.chaanti2out], ]; typeEntries.forEach(function(te) { var typeName = te[0], machine = te[1], inp = te[2], out2 = te[3]; if (!typeName) return; var ts = ss.getSheetByName(typeName); if (!ts) { ts = ss.insertSheet(typeName); ts.appendRow(['Date','Time','Machine','Input','Output']); } ts.appendRow([r.date, r.time, machine, inp, out2]); }); } // Delete from main sheet by row match (date+time) if (data.action === 'delete') { var rows = main.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.date) && String(rows[i][1]) === String(data.time)) { main.deleteRow(i + 1); break; } } } return out({ status: 'ok' }); } function out(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
